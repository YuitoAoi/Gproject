```python
#!/usr/bin/env python3
"""
Phase 2 Data Management Module - End-to-End Verification Script

Usage:
    python verify_data_management.py [--base-url http://localhost:8000]

Prerequisites:
    - Backend running (uvicorn)
    - MySQL and Redis running
    - Celery worker running
"""

import argparse
import requests
import time
import os
import sys

BASE_URL = "http://localhost:8000"
API_PREFIX = "/api/v1"

def log(msg, status="INFO"):
    print(f"[{status}] {msg}")

def test_upload_flow():
    """Test complete upload flow: initiate -> upload chunks -> complete"""
    log("Testing upload flow...")
    
    # Create a small test file
    test_content = "name,age,city\nAlice,30,Beijing\nBob,25,Shanghai"
    test_file_path = "/tmp/test_dataset.csv"
    with open(test_file_path, 'w') as f:
        f.write(test_content)
    
    file_size = os.path.getsize(test_file_path)
    
    # 1. Initiate upload
    resp = requests.get(
        f"{BASE_URL}{API_PREFIX}/datasets/initiate-upload",
        params={"filename": "test_dataset.csv", "file_size": file_size, "format": "csv"}
    )
    assert resp.status_code == 200, f"Initiate failed: {resp.text}"
    data = resp.json()
    upload_id = data["upload_id"]
    chunk_size = data["chunk_size"]
    total_chunks = data["total_chunks"]
    log(f"Upload initiated: {upload_id}, chunks: {total_chunks}")
    
    # 2. Upload chunks
    with open(test_file_path, 'rb') as f:
        for i in range(total_chunks):
            chunk = f.read(chunk_size)
            files = {'file': ('chunk', chunk, 'application/octet-stream')}
            resp = requests.post(
                f"{BASE_URL}{API_PREFIX}/datasets/upload-chunk",
                params={"upload_id": upload_id, "chunk_number": i},
                files=files
            )
            assert resp.status_code == 200, f"Chunk {i} upload failed: {resp.text}"
            log(f"Chunk {i} uploaded")
    
    # 3. Complete upload
    resp = requests.post(
        f"{BASE_URL}{API_PREFIX}/datasets/complete-upload",
        params={"upload_id": upload_id, "filename": "test_dataset.csv", "format": "csv", "file_size": file_size}
    )
    assert resp.status_code == 200, f"Complete failed: {resp.text}"
    log("Upload complete, waiting for Celery task...")
    
    # Wait for task to complete (check task status)
    time.sleep(5)
    
    # 4. Verify dataset was created
    resp = requests.get(f"{BASE_URL}{API_PREFIX}/datasets")
    assert resp.status_code == 200, f"List datasets failed: {resp.text}"
    datasets = resp.json()
    assert len(datasets) > 0, "No datasets found after upload"
    log(f"Dataset created: {datasets[0]['name']}")
    
    return datasets[0]["id"]

def test_process_flow(dataset_id):
    """Test data processing flow"""
    log("Testing processing flow...")
    
    # Trigger cleaning
    resp = requests.post(
        f"{BASE_URL}{API_PREFIX}/datasets/{dataset_id}/process",
        json={"process_type": "clean", "remove_duplicates": True}
    )
    assert resp.status_code == 200, f"Process failed: {resp.text}"
    task_id = resp.json()["task_id"]
    log(f"Cleaning task started: {task_id}")
    
    # Wait and check status
    time.sleep(3)
    resp = requests.get(f"{BASE_URL}{API_PREFIX}/datasets/tasks/{task_id}")
    log(f"Task status: {resp.json()['status']}")

def test_delete_flow(dataset_id):
    """Test delete flow"""
    log("Testing delete flow...")
    
    resp = requests.delete(f"{BASE_URL}{API_PREFIX}/datasets/{dataset_id}")
    assert resp.status_code == 204, f"Delete failed: {resp.status_code}"
    log("Dataset deleted")
    
    # Verify deletion
    resp = requests.get(f"{BASE_URL}{API_PREFIX}/datasets/{dataset_id}")
    assert resp.status_code == 404, "Dataset should not exist after deletion"

def main():
    parser = argparse.ArgumentParser(description="Verify Data Management Module")
    parser.add_argument("--base-url", default=BASE_URL, help="Backend base URL")
    args = parser.parse_args()
    
    global BASE_URL
    BASE_URL = args.base_url.rstrip("/")
    
    log("=" * 50)
    log("Phase 2 Data Management - E2E Verification")
    log("=" * 50)
    
    try:
        # Test 1: Upload flow
        dataset_id = test_upload_flow()
        
        # Test 2: Process flow
        test_process_flow(dataset_id)
        
        # Test 3: Delete flow
        test_delete_flow(dataset_id)
        
        log("=" * 50)
        log("ALL TESTS PASSED", "SUCCESS")
        log("=" * 50)
        return 0
        
    except AssertionError as e:
        log(f"TEST FAILED: {e}", "ERROR")
        return 1
    except Exception as e:
        log(f"ERROR: {e}", "ERROR")
        return 1

if __name__ == "__main__":
    sys.exit(main())
```
