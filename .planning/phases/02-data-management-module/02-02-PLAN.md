import httpClient from './httpClient'
import type { AxiosProgressEvent } from 'axios'

// Types
export interface Dataset {
  id: number
  name: string
  description: string | null
  file_path: string
  file_size: number
  format: string
  total_records: number
  status: 'pending' | 'processing' | 'converting' | 'ready' | 'error'
  created_at: string
  updated_at: string
}

export interface UploadInitiateResponse {
  upload_id: string
  chunk_size: number
  total_chunks: number
}

export interface UploadChunkResponse {
  upload_id: string
  chunk_number: number
  received: boolean
  message: string
}

export interface UploadCompleteResponse {
  upload_id: string
  dataset_id: number
  status: string
  message: string
}

export interface ProcessRequest {
  process_type: 'clean' | 'convert'
  convert_format?: 'alpaca' | 'sharegpt'
  remove_duplicates?: boolean
  fill_missing?: boolean
  missing_strategy?: 'mean' | 'median' | 'drop'
}

export interface ProcessResponse {
  task_id: string
  status: string
  message: string
}

export interface TaskStatusResponse {
  task_id: string
  status: string
  result?: Record<string, any>
  error?: string
}

const CHUNK_SIZE = 5 * 1024 * 1024 // 5MB

/**
 * 启动分块上传
 */
export async function initiateUpload(filename: string, fileSize: number, format: string): Promise<UploadInitiateResponse> {
  const response = await httpClient.get<UploadInitiateResponse>('/datasets/initiate-upload', {
    params: { filename, file_size: fileSize, format }
  })
  return response.data
}

/**
 * 上传单个分块
 */
export async function uploadChunk(
  uploadId: string,
  chunkNumber: number,
  chunk: Blob,
  onProgress?: (percent: number) => void
): Promise<UploadChunkResponse> {
  const formData = new FormData()
  formData.append('file', chunk)
  
  const response = await httpClient.post<UploadChunkResponse>(
    `/datasets/upload-chunk?upload_id=${uploadId}&chunk_number=${chunkNumber}`,
    formData,
    {
      headers: { 'Content-Type': 'multipart/form-data' },
      onUploadProgress: (progressEvent: AxiosProgressEvent) => {
        if (progressEvent.total && onProgress) {
          onProgress(Math.round((progressEvent.loaded * 100) / progressEvent.total))
        }
      }
    }
  )
  return response.data
}

/**
 * 完成上传
 */
export async function completeUpload(uploadId: string, filename: string, format: string, fileSize: number): Promise<UploadCompleteResponse> {
  const response = await httpClient.post<UploadCompleteResponse>('/datasets/complete-upload', null, {
    params: { upload_id: uploadId, filename, format, file_size: fileSize }
  })
  return response.data
}

/**
 * 完整的分块上传流程
 */
export async function uploadDataset(
  file: File,
  onProgress?: (percent: number, phase: string) => void
): Promise<UploadCompleteResponse> {
  const format = file.name.split('.').pop() || 'csv'
  
  // 1. 启动上传
  onProgress?.(0, 'initiating')
  const initResponse = await initiateUpload(file.name, file.size, format)
  const { upload_id, chunk_size, total_chunks } = initResponse
  
  // 2. 上传分块
  const chunks: Blob[] = []
  let start = 0
  let chunkNum = 0
  while (start < file.size) {
    const end = Math.min(start + chunk_size, file.size)
    chunks.push(file.slice(start, end))
    start = end
    chunkNum++
  }
  
  // 并发上传分块（最多3个同时）
  const chunkProgress: number[] = new Array(total_chunks).fill(0)
  let completedChunks = 0
  
  const uploadChunkWithProgress = async (chunk: Blob, num: number): Promise<void> => {
    await uploadChunk(upload_id, num, chunk, (percent) => {
      chunkProgress[num] = percent
      const avgProgress = chunkProgress.reduce((a, b) => a + b, 0) / total_chunks
      onProgress?.(Math.round(avgProgress), 'uploading')
    })
    completedChunks++
  }
  
  // 分批并发上传
  const batchSize = 3
  for (let i = 0; i < chunks.length; i += batchSize) {
    const batch = chunks.slice(i, i + batchSize)
    await Promise.all(batch.map((chunk, idx) => uploadChunkWithProgress(chunk, i + idx)))
  }
  
  // 3. 完成上传
  onProgress?.(100, 'completing')
  const completeResponse = await completeUpload(upload_id, file.name, format, file.size)
  
  onProgress?.(100, 'complete')
  return completeResponse
}

/**
 * 获取数据集列表
 */
export async function getDatasets(): Promise<Dataset[]> {
  const response = await httpClient.get<Dataset[]>('/datasets')
  return response.data
}

/**
 * 获取单个数据集
 */
export async function getDataset(id: number): Promise<Dataset> {
  const response = await httpClient.get<Dataset>(`/datasets/${id}`)
  return response.data
}

/**
 * 删除数据集
 */
export async function deleteDataset(id: number): Promise<void> {
  await httpClient.delete(`/datasets/${id}`)
}

/**
 * 触发数据处理任务
 */
export async function processDataset(datasetId: number, request: ProcessRequest): Promise<ProcessResponse> {
  const response = await httpClient.post<ProcessResponse>(
    `/datasets/${datasetId}/process`,
    request
  )
  return response.data
}

/**
 * 获取任务状态
 */
export async function getTaskStatus(taskId: string): Promise<TaskStatusResponse> {
  const response = await httpClient.get<TaskStatusResponse>(`/datasets/tasks/${taskId}`)
  return response.data
}

<script setup lang="ts">
import { ref, onMounted, computed } from 'vue'
import { ElMessage, ElMessageBox } from 'element-plus'
import type { Dataset, ProcessRequest } from '@/api/dataset'
import { getDatasets, deleteDataset, processDataset, getTaskStatus } from '@/api/dataset'

const emit = defineEmits<{
  (e: 'refresh'): void
}>()

const datasets = ref<Dataset[]>([])
const loading = ref(false)
const processingTasks = ref<Map<number, string>>(new Map()) // datasetId -> taskId

const statusType = (status: string) => {
  const map: Record<string, string> = {
    ready: 'success',
    processing: 'warning',
    converting: 'warning',
    pending: 'info',
    error: 'danger'
  }
  return map[status] || 'info'
}

const formatSize = (bytes: number) => {
  if (bytes < 1024) return bytes + ' B'
  if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB'
  return (bytes / (1024 * 1024)).toFixed(1) + ' MB'
}

const formatDate = (dateStr: string) => {
  return new Date(dateStr).toLocaleString()
}

const canProcess = (dataset: Dataset) => {
  return dataset.status === 'ready'
}

const handleDelete = async (dataset: Dataset) => {
  try {
    await ElMessageBox.confirm(
      `确定要删除数据集 "${dataset.name}" 吗？此操作将同时删除关联的文件。`,
      '确认删除',
      { type: 'warning' }
    )
    
    await deleteDataset(dataset.id)
    ElMessage.success('删除成功')
    emit('refresh')
  } catch (err: any) {
    if (err !== 'cancel') {
      ElMessage.error('删除失败: ' + (err.message || '未知错误'))
    }
  }
}

const handleProcess = async (dataset: Dataset, type: 'clean' | 'convert', format?: 'alpaca' | 'sharegpt') => {
  const request: ProcessRequest = {
    process_type: type
  }
  
  if (type === 'convert' && format) {
    request.convert_format = format
  }
  
  try {
    const response = await processDataset(dataset.id, request)
    processingTasks.value.set(dataset.id, response.task_id)
    ElMessage.success('处理任务已启动')
    
    // 轮询任务状态
    pollTaskStatus(dataset.id, response.task_id)
  } catch (err: any) {
    ElMessage.error('启动处理失败: ' + (err.message || '未知错误'))
  }
}

const pollTaskStatus = async (datasetId: number, taskId: string) => {
  const poll = async () => {
    try {
      const status = await getTaskStatus(taskId)
      if (status.status === 'SUCCESS' || status.status === 'FAILURE') {
        processingTasks.value.delete(datasetId)
        emit('refresh')
        ElMessage.success(status.status === 'SUCCESS' ? '处理完成' : '处理失败')
      } else {
        setTimeout(poll, 2000)
      }
    } catch {
      setTimeout(poll, 2000)
    }
  }
  poll()
}

const loadDatasets = async () => {
  loading.value = true
  try {
    datasets.value = await getDatasets()
  } catch (err: any) {
    ElMessage.error('加载数据集失败: ' + (err.message || '未知错误'))
  } finally {
    loading.value = false
  }
}

onMounted(() => {
  loadDatasets()
})

defineExpose({ refresh: loadDatasets })
</script>

import { ref, computed } from 'vue'
import { ElMessage, ElUpload, ElButton, ElProgress, ElIcon } from 'element-plus'
import { UploadFilled } from '@element-plus/icons-vue'
import { uploadDataset } from '@/api/dataset'

const emit = defineEmits<{
  (e: 'upload-complete'): void
}>()

const uploading = ref(false)
const uploadProgress = ref(0)
const uploadPhase = ref('')
const uploadError = ref('')
const currentFile = ref<File | null>(null)

const phaseText = computed(() => {
  const map: Record<string, string> = {
    initiating: '准备上传...',
    uploading: '上传中...',
    completing: '处理中...',
    complete: '上传完成',
    error: '上传失败'
  }
  return map[uploadPhase.value] || ''
})

const formatSize = (bytes: number) => {
  if (bytes < 1024) return bytes + ' B'
  if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB'
  return (bytes / (1024 * 1024)).toFixed(1) + ' MB'
}

const handleFileChange = (file: File) => {
  // 验证文件类型
  const ext = file.name.split('.').pop()?.toLowerCase()
  if (!['csv', 'xlsx'].includes(ext || '')) {
    ElMessage.error('仅支持 CSV 和 XLSX 格式文件')
    return false
  }
  
  // 验证文件大小 (100MB max)
  if (file.size > 100 * 1024 * 1024) {
    ElMessage.error('文件大小不能超过 100MB')
    return false
  }
  
  currentFile.value = file
  uploadError.value = ''
  return true
}

const handleUpload = async () => {
  if (!currentFile.value) {
    ElMessage.warning('请先选择文件')
    return
  }
  
  uploading.value = true
  uploadProgress.value = 0
  uploadError.value = ''
  
  try {
    await uploadDataset(currentFile.value, (percent, phase) => {
      uploadProgress.value = percent
      uploadPhase.value = phase
    })
    
    uploadPhase.value = 'complete'
    ElMessage.success('上传成功')
    emit('upload-complete')
    
    // 3秒后重置状态
    setTimeout(() => {
      resetUpload()
    }, 3000)
  } catch (err: any) {
    uploadPhase.value = 'error'
    uploadError.value = err.response?.data?.detail || err.message || '上传失败'
    ElMessage.error('上传失败: ' + uploadError.value)
  } finally {
    uploading.value = false
  }
}

const resetUpload = () => {
  uploading.value = false
  uploadProgress.value = 0
  uploadPhase.value = ''
  uploadError.value = ''
  currentFile.value = null
}

const retryUpload = () => {
  if (currentFile.value) {
    handleUpload()
  }
}
</script>

<template>

        :show-file-list="false"
