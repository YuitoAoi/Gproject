```python
from pydantic import BaseModel, Field, field_validator
from typing import Optional, Literal
from datetime import datetime


class DatasetCreate(BaseModel):
    """创建数据集请求"""
    name: str = Field(..., min_length=1, max_length=100, description="数据集名称")
    description: Optional[str] = Field(None, max_length=500, description="数据集描述")
    format: Literal["csv", "xlsx", "json"] = Field(..., description="数据格式")


class DatasetResponse(BaseModel):
    """数据集响应"""
    id: int
    name: str
    description: Optional[str]
    file_path: str
    file_size: int
    format: str
    total_records: int
    status: str
    created_at: datetime
    updated_at: datetime

    class Config:
        from_attributes = True


class UploadInitiateResponse(BaseModel):
    """启动分块上传响应"""
    upload_id: str = Field(..., description="唯一上传ID")
    chunk_size: int = Field(5 * 1024 * 1024, description="分块大小 (5MB)")
    total_chunks: int = Field(..., description="总分块数")


class UploadChunkResponse(BaseModel):
    """上传分块响应"""
    upload_id: str
    chunk_number: int
    received: bool
    message: str


class UploadCompleteResponse(BaseModel):
    """完成上传响应"""
    upload_id: str
    dataset_id: int
    status: str
    message: str


class ProcessRequest(BaseModel):
    """数据处理请求"""
    process_type: Literal["clean", "convert"] = Field(..., description="处理类型")
    convert_format: Optional[Literal["alpaca", "sharegpt"]] = Field(None, description="转换目标格式")
    # 清洗选项
    remove_duplicates: bool = Field(False, description="删除重复行")
    fill_missing: bool = Field(False, description="填充缺失值")
    missing_strategy: Optional[Literal["mean", "median", "drop"]] = Field(None, description="缺失值策略")


class ProcessResponse(BaseModel):
    """处理任务响应"""
    task_id: str
    status: str
    message: str


class TaskStatusResponse(BaseModel):
    """任务状态响应"""
    task_id: str
    status: str
    result: Optional[dict] = None
    error: Optional[str] = None
```

```python
from sqlalchemy.orm import Session
from typing import List, Optional
from app.db.models import Dataset
from app.schemas.dataset import DatasetCreate


def get_datasets(db: Session, skip: int = 0, limit: int = 100) -> List[Dataset]:
    """获取所有数据集"""
    return db.query(Dataset).offset(skip).limit(limit).all()


def get_dataset(db: Session, dataset_id: int) -> Optional[Dataset]:
    """获取单个数据集"""
    return db.query(Dataset).filter(Dataset.id == dataset_id).first()


def create_dataset(db: Session, dataset_in: DatasetCreate, file_path: str, file_size: int, owner_id: int = 1) -> Dataset:
    """创建数据集记录"""
    dataset = Dataset(
        name=dataset_in.name,
        description=dataset_in.description,
        file_path=file_path,
        file_size=file_size,
        format=dataset_in.format,
        status="ready",
        owner_id=owner_id
    )
    db.add(dataset)
    db.commit()
    db.refresh(dataset)
    return dataset


def delete_dataset(db: Session, dataset_id: int) -> bool:
    """删除数据集"""
    dataset = db.query(Dataset).filter(Dataset.id == dataset_id).first()
    if dataset:
        db.delete(dataset)
        db.commit()
        return True
    return False


def update_dataset_status(db: Session, dataset_id: int, status: str, total_records: int = None) -> Optional[Dataset]:
    """更新数据集状态"""
    dataset = db.query(Dataset).filter(Dataset.id == dataset_id).first()
    if dataset:
        dataset.status = status
        if total_records is not None:
            dataset.total_records = total_records
        db.commit()
        db.refresh(dataset)
    return dataset
```

```python
import os
import uuid
import aiofiles
from fastapi import APIRouter, Depends, HTTPException, UploadFile, File, Query
from sqlalchemy.orm import Session
from celery.result import AsyncResult

from app.db.session import get_db
from app.schemas.dataset import (
    DatasetCreate, DatasetResponse, UploadInitiateResponse, UploadChunkResponse,
    UploadCompleteResponse, ProcessRequest, ProcessResponse, TaskStatusResponse
)
from app.crud import dataset as crud
from app.tasks.dataset_tasks import assemble_and_save_dataset, process_dataset_clean, convert_dataset_format

router = APIRouter()

CHUNK_SIZE = 5 * 1024 * 1024  # 5MB chunks
UPLOAD_DIR = os.path.join(os.path.dirname(os.path.dirname(os.path.dirname(__file__))), "uploads", "chunks")


def ensure_upload_dir():
    os.makedirs(UPLOAD_DIR, exist_ok=True)


@router.post("/datasets/initiate-upload", response_model=UploadInitiateResponse)
async def initiate_upload(filename: str = Query(...), file_size: int = Query(...), format: str = Query(...)):
    """启动分块上传流程"""
    ensure_upload_dir()
    upload_id = str(uuid.uuid4())
    total_chunks = (file_size + CHUNK_SIZE - 1) // CHUNK_SIZE
    return UploadInitiateResponse(
        upload_id=upload_id,
        chunk_size=CHUNK_SIZE,
        total_chunks=total_chunks
    )


@router.post("/datasets/upload-chunk", response_model=UploadChunkResponse)
async def upload_chunk(upload_id: str = Query(...), chunk_number: int = Query(...), file: UploadFile = File(...)):
    """上传单个文件分块"""
    ensure_upload_dir()
    chunk_path = os.path.join(UPLOAD_DIR, f"{upload_id}_chunk_{chunk_number}")
    
    async with aiofiles.open(chunk_path, 'wb') as f:
        content = await file.read()
        await f.write(content)
    
    return UploadChunkResponse(
        upload_id=upload_id,
        chunk_number=chunk_number,
        received=True,
        message=f"Chunk {chunk_number} received"
    )


@router.post("/datasets/complete-upload", response_model=UploadCompleteResponse)
async def complete_upload(upload_id: str = Query(...), filename: str = Query(...), format: str = Query(...), file_size: int = Query(...)):
    """完成分块上传并触发文件合并"""
    # 触发 Celery 任务来合并文件
    task = assemble_and_save_dataset.delay(upload_id, filename, format, file_size)
    
    return UploadCompleteResponse(
        upload_id=upload_id,
        dataset_id=0,  # 暂时设为0，任务完成后会更新
        status="processing",
        message=f"File assembly task started: {task.id}"
    )


@router.get("/datasets", response_model=list[DatasetResponse])
async def list_datasets(skip: int = Query(0, ge=0), limit: int = Query(100, ge=1, le=1000), db: Session = Depends(get_db)):
    """获取数据集列表"""
    datasets = crud.get_datasets(db, skip=skip, limit=limit)
    return datasets


@router.get("/datasets/{dataset_id}", response_model=DatasetResponse)
async def get_dataset(dataset_id: int, db: Session = Depends(get_db)):
    """获取单个数据集"""
    dataset = crud.get_dataset(db, dataset_id)
    if not dataset:
        raise HTTPException(status_code=404, detail="Dataset not found")
    return dataset


@router.delete("/datasets/{dataset_id}", status_code=204)
async def delete_dataset(dataset_id: int, db: Session = Depends(get_db)):
    """删除数据集"""
    dataset = crud.get_dataset(db, dataset_id)
    if not dataset:
        raise HTTPException(status_code=404, detail="Dataset not found")
    
    # 删除文件
    if os.path.exists(dataset.file_path):
        os.remove(dataset.file_path)
    
    # 删除数据库记录
    crud.delete_dataset(db, dataset_id)
    return None


@router.post("/datasets/{dataset_id}/process", response_model=ProcessResponse)
async def process_dataset(dataset_id: int, request: ProcessRequest, db: Session = Depends(get_db)):
    """触发数据处理任务（清洗或转换）"""
    dataset = crud.get_dataset(db, dataset_id)
    if not dataset:
        raise HTTPException(status_code=404, detail="Dataset not found")
    
    if request.process_type == "clean":
        task = process_dataset_clean.delay(
            dataset_id=dataset_id,
            remove_duplicates=request.remove_duplicates,
            fill_missing=request.fill_missing,
            missing_strategy=request.missing_strategy
        )
    elif request.process_type == "convert":
        if not request.convert_format:
            raise HTTPException(status_code=400, detail="convert_format is required for conversion")
        task = convert_dataset_format.delay(
            dataset_id=dataset_id,
            target_format=request.convert_format
        )
    else:
        raise HTTPException(status_code=400, detail="Invalid process_type")
    
    return ProcessResponse(
        task_id=task.id,
        status="pending",
        message=f"{request.process_type} task started"
    )


@router.get("/datasets/tasks/{task_id}", response_model=TaskStatusResponse)
async def get_task_status(task_id: str):
    """获取异步任务状态"""
    task_result = AsyncResult(task_id)
    response = TaskStatusResponse(
        task_id=task_id,
        status=task_result.status
    )
    if task_result.ready():
        if task_result.successful():
            response.result = task_result.result
        else:
            response.error = str(task_result.result)
    return response
```

```python
import os
import json
import pandas as pd
from app.core.celery_app import celery_app
from app.db.session import SessionLocal
from app.crud.dataset import update_dataset_status, create_dataset, get_dataset
from app.schemas.dataset import DatasetCreate

UPLOAD_DIR = os.path.join(os.path.dirname(os.path.dirname(os.path.dirname(__file__))), "uploads")
DATASET_DIR = os.path.join(os.path.dirname(os.path.dirname(os.path.dirname(__file__))), "datasets")


def ensure_dirs():
    os.makedirs(UPLOAD_DIR, exist_ok=True)
    os.makedirs(DATASET_DIR, exist_ok=True)


@celery_app.task(name='app.tasks.dataset_tasks.assemble_and_save_dataset')
def assemble_and_save_dataset(upload_id: str, filename: str, format: str, file_size: int) -> dict:
    """合并分块文件并创建数据集记录"""
    ensure_dirs()
    db = SessionLocal()
    try:
        upload_dir = os.path.join(UPLOAD_DIR, "chunks")
        
        # 查找所有分块
        chunk_files = sorted([
            f for f in os.listdir(upload_dir) 
            if f.startswith(f"{upload_id}_chunk_")
        ], key=lambda x: int(x.split("_chunk_")[1]))
        
        if not chunk_files:
            raise Exception("No chunk files found")
        
        # 合并文件
        final_path = os.path.join(DATASET_DIR, filename)
        with open(final_path, 'wb') as outfile:
            for chunk_file in chunk_files:
                chunk_path = os.path.join(upload_dir, chunk_file)
                with open(chunk_path, 'rb') as infile:
                    outfile.write(infile.read())
                os.remove(chunk_path)  # 清理分块
        
        # 创建数据库记录
        dataset_in = DatasetCreate(name=filename.rsplit('.', 1)[0], description="", format=format)
        dataset = create_dataset(db, dataset_in, final_path, file_size)
        
        # 清理 Redis upload_id (如果使用了 Redis 存储)
        
        return {"dataset_id": dataset.id, "status": "ready"}
    finally:
        db.close()


@celery_app.task(name='app.tasks.dataset_tasks.process_dataset_clean')
def process_dataset_clean(dataset_id: int, remove_duplicates: bool = False, fill_missing: bool = False, missing_strategy: str = None) -> dict:
    """数据清洗任务"""
    db = SessionLocal()
    try:
        dataset = get_dataset(db, dataset_id)
        if not dataset:
            raise Exception(f"Dataset {dataset_id} not found")
        
        update_dataset_status(db, dataset_id, "processing")
        
        # 读取数据
        if dataset.format == "csv":
            df = pd.read_csv(dataset.file_path)
        elif dataset.format == "xlsx":
            df = pd.read_excel(dataset.file_path)
        else:
            raise Exception(f"Unsupported format: {dataset.format}")
        
        original_rows = len(df)
        
        # 去除重复
        if remove_duplicates:
            df = df.drop_duplicates()
        
        # 处理缺失值
        if fill_missing and missing_strategy:
            if missing_strategy == "mean":
                df = df.fillna(df.mean(numeric_only=True))
            elif missing_strategy == "median":
                df = df.fillna(df.median(numeric_only=True))
            elif missing_strategy == "drop":
                df = df.dropna()
        
        # 保存清洗后的数据
        cleaned_path = dataset.file_path.replace(dataset.name, f"{dataset.name}_cleaned")
        if dataset.format == "csv":
            df.to_csv(cleaned_path, index=False)
        elif dataset.format == "xlsx":
            df.to_excel(cleaned_path, index=False)
        
        # 更新数据库
        update_dataset_status(db, dataset_id, "ready", len(df))
        
        return {
            "dataset_id": dataset_id,
            "original_rows": original_rows,
            "cleaned_rows": len(df),
            "duplicates_removed": original_rows - len(df.drop_duplicates()) if remove_duplicates else 0
        }
    finally:
        db.close()


@celery_app.task(name='app.tasks.dataset_tasks.convert_dataset_format')
def convert_dataset_format(dataset_id: int, target_format: str) -> dict:
    """数据集格式转换任务"""
    db = SessionLocal()
    try:
        dataset = get_dataset(db, dataset_id)
        if not dataset:
            raise Exception(f"Dataset {dataset_id} not found")
        
        update_dataset_status(db, dataset_id, "converting")
        
        # 读取源数据
        if dataset.format == "csv":
            df = pd.read_csv(dataset.file_path)
        elif dataset.format == "xlsx":
            df = pd.read_excel(dataset.file_path)
        else:
            raise Exception(f"Unsupported source format: {dataset.format}")
        
        # 转换为目标格式
        base_name = os.path.splitext(os.path.basename(dataset.file_path))[0]
        
        if target_format == "alpaca":
            # Alpaca 格式: {"instruction": "...", "input": "...", "output": "..."}
            alpaca_data = []
            for _, row in df.iterrows():
                # 假设第一列是 instruction, 第二列是 output
                cols = df.columns.tolist()
                alpaca_item = {
                    "instruction": str(row.get(cols[0], "")),
                    "input": "",
                    "output": str(row.get(cols[1] if len(cols) > 1 else cols[0], ""))
                }
                alpaca_data.append(alpaca_item)
            
            output_path = os.path.join(DATASET_DIR, f"{base_name}_alpaca.json")
            with open(output_path, 'w', encoding='utf-8') as f:
                json.dump(alpaca_data, f, ensure_ascii=False, indent=2)
            
            # 生成 dataset_info.json
            info_path = os.path.join(DATASET_DIR, f"{base_name}_alpaca_dataset_info.json")
            info = {
                "file_name": f"{base_name}_alpaca.json",
                "format": "alpaca",
                "columns": {"instruction": "str", "input": "str", "output": "str"},
                "num_samples": len(alpaca_data)
            }
            with open(info_path, 'w', encoding='utf-8') as f:
                json.dump(info, f, ensure_ascii=False, indent=2)
        
        elif target_format == "sharegpt":
            # ShareGPT 格式: {"conversations": [{"from": "...", "value": "..."}]}
            sharegpt_data = []
            for _, row in df.iterrows():
                cols = df.columns.tolist()
                conversation = [
                    {"from": "human", "value": str(row.get(cols[0], ""))},
                    {"from": "gpt", "value": str(row.get(cols[1] if len(cols) > 1 else cols[0], ""))}
                ]
                sharegpt_data.append({"conversations": conversation})
            
            output_path = os.path.join(DATASET_DIR, f"{base_name}_sharegpt.json")
            with open(output_path, 'w', encoding='utf-8') as f:
                json.dump(sharegpt_data, f, ensure_ascii=False, indent=2)
            
            # 生成 dataset_info.json
            info_path = os.path.join(DATASET_DIR, f"{base_name}_sharegpt_dataset_info.json")
            info = {
                "file_name": f"{base_name}_sharegpt.json",
                "format": "sharegpt",
                "num_samples": len(sharegpt_data)
            }
            with open(info_path, 'w', encoding='utf-8') as f:
                json.dump(info, f, ensure_ascii=False, indent=2)
        else:
            raise Exception(f"Unsupported target format: {target_format}")
        
        update_dataset_status(db, dataset_id, "ready")
        
        return {
            "dataset_id": dataset_id,
            "target_format": target_format,
            "output_path": output_path,
            "samples_count": len(df)
        }
    finally:
        db.close()
```
